```typescript
export const summarySchema = z.object({
  generated_at: z.iso.datetime(),
  period: z.object({ since: z.string(), until: z.string(), days: z.number() }),
  stats: z.object({
    total_prompts: z.number(),
    total_tool_uses: z.number(),
    total_permissions: z.number(),
    unique_sessions: z.number(),
  }),
  top_repeated_prompts: z.array(z.object({
    prompt: z.string(), count: z.number(), sessions: z.number(),
  })).max(20),
  tool_frequency: z.array(z.object({
    tool_name: z.string(), count: z.number(), avg_duration_ms: z.number().optional(),
  })),
  permission_patterns: z.array(z.object({
    tool_name: z.string(), count: z.number(), sessions: z.number(),
  })),
  long_prompts: z.array(z.object({
    prompt_preview: z.string(), length: z.number(), count: z.number(),
  })).max(10),
});
export type Summary = z.infer<typeof summarySchema>;
```

    ```
    import { readLogEntries } from './jsonl-reader.js';
    import { summarySchema, type Summary } from './schemas.js';
    import { promptEntrySchema, toolEntrySchema, permissionEntrySchema } from '../schemas/log-entry.js';
    import type { PromptEntry, ToolEntry, PermissionEntry } from '../schemas/log-entry.js';
    import { paths, ensureInit } from '../storage/dirs.js';
    import writeFileAtomic from 'write-file-atomic';
    ```

    Constants:
    - `PROMPT_TRUNCATE_LEN = 100`
    - `LONG_PROMPT_THRESHOLD = 200` (word count)
    - `DEFAULT_TOP_N = 20`
    - `DEFAULT_DAYS = 30`
    - `MAX_LONG_PROMPTS = 10`

    Export one main function:
    ```typescript
    export async function preProcess(options?: {
      since?: Date;
      until?: Date;
      topN?: number;
    }): Promise<Summary>
    ```

    Implementation steps:
    1. Compute date range: until defaults to `new Date()`, since defaults to `until - DEFAULT_DAYS * 86400000`
    2. Call `readLogEntries(paths.logs.prompts, promptEntrySchema, { since, until })` for prompts
    3. Call `readLogEntries(paths.logs.tools, toolEntrySchema, { since, until })` for tools
    4. Call `readLogEntries(paths.logs.permissions, permissionEntrySchema, { since, until })` for permissions
    5. Compute unique sessions across all three entry arrays using a Set<string>
    6. **Prompt frequency** (internal helper `countWithSessions`):
       - Use `Map<string, { count: number; sessions: Set<string> }>` 
       - Key = `normalizePrompt(entry.prompt)` -- trim, lowercase, collapse `\s+` to single space
       - For each prompt entry, increment count and add session_id to sessions Set
       - Extract top-N: sort by count descending, slice(0, topN)
       - Map to output format: { prompt: key.slice(0, PROMPT_TRUNCATE_LEN), count, sessions: set.size }
    7. **Tool frequency** (internal helper `computeToolFrequency`):
       - Use `Map<string, { count: number; durations: number[] }>` 
       - For ALL tool entries: increment count for tool_name
       - For 'post' events with duration_ms: push duration to durations array
       - Map to output: { tool_name, count, avg_duration_ms: durations.length > 0 ? Math.round(sum/len) : undefined }
       - Sort by count descending
    8. **Permission frequency** (reuse `countWithSessions`):
       - Key = entry.tool_name, session = entry.session_id
       - Extract all (no top-N limit for permissions -- typically fewer unique tools)
       - Map to output: { tool_name: key, count, sessions: set.size }
       - Sort by count descending
    9. **Long prompts** (internal helper `detectLongPrompts`):
       - Filter prompts where `entry.prompt.trim().split(/\s+/).length > LONG_PROMPT_THRESHOLD`
       - Group by normalizePrompt(entry.prompt), count occurrences
       - Sort by count descending, take first MAX_LONG_PROMPTS
       - Map to output: { prompt_preview: normalized.slice(0, PROMPT_TRUNCATE_LEN), length: word count, count }
    10. Construct Summary object, validate with `summarySchema.parse(summary)`
    11. Write atomically: `await ensureInit(); await writeFileAtomic(paths.summary, JSON.stringify(summary));`
    12. Return the validated Summary

    Internal helpers (NOT exported -- all within the module):
    - `normalizePrompt(prompt: string): string` -- trim, toLowerCase, replace(/\s+/g, ' ')
    - `countWithSessions(items: Array<{ key: string; session: string }>): Map<string, { count: number; sessions: Set<string> }>`
    - `computeToolFrequency(tools: ToolEntry[]): Summary['tool_frequency']`
    - `detectLongPrompts(prompts: PromptEntry[]): Summary['long_prompts']`
    - `formatDate(d: Date): string` -- d.toISOString().slice(0, 10)
