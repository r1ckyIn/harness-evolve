    2. **src/scrubber/scrub.ts** — Core scrubbing functions:
       ```typescript
       import { SCRUB_PATTERNS, type ScrubPattern } from './patterns.js';

       export function scrubString(input: string, extraPatterns?: ScrubPattern[]): string {
         let result = input;
         const patterns = extraPatterns ? [...SCRUB_PATTERNS, ...extraPatterns] : SCRUB_PATTERNS;
         for (const pattern of patterns) {
           // Reset regex lastIndex for global patterns
           pattern.regex.lastIndex = 0;
           result = result.replace(pattern.regex, pattern.replacement);
         }
         return result;
       }

       export function scrubObject<T>(obj: T, extraPatterns?: ScrubPattern[]): T {
         if (typeof obj === 'string') {
           return scrubString(obj, extraPatterns) as T;
         }
         if (Array.isArray(obj)) {
           return obj.map(item => scrubObject(item, extraPatterns)) as T;
         }
         if (obj !== null && typeof obj === 'object') {
           const result: Record<string, unknown> = {};
           for (const [key, value] of Object.entries(obj)) {
             result[key] = scrubObject(value, extraPatterns);
           }
           return result as T;
         }
         return obj; // numbers, booleans, null, undefined pass through
       }
       ```

       CRITICAL: Each regex pattern uses the `/g` (global) flag. When reusing regex objects, `lastIndex` must be reset to 0 before each `replace()` call. The `replace()` method on strings already handles this correctly when called with a global regex, but explicitly resetting is defensive.

    1. **src/storage/logger.ts** — JSONL logger implementing the scrub-before-write pipeline:
       ```typescript
       import { appendFile } from 'node:fs/promises';
       import { join } from 'node:path';
       import { z } from 'zod/v4';
       import { promptEntrySchema, toolEntrySchema, permissionEntrySchema, sessionEntrySchema } from '../schemas/log-entry.js';
       import { scrubObject } from '../scrubber/scrub.js';
       import { paths, ensureInit } from './dirs.js';

       // Map log types to their schemas and directories
       const LOG_TYPE_MAP = {
         prompts: { schema: promptEntrySchema, dir: paths.logs.prompts },
         tools: { schema: toolEntrySchema, dir: paths.logs.tools },
         permissions: { schema: permissionEntrySchema, dir: paths.logs.permissions },
         sessions: { schema: sessionEntrySchema, dir: paths.logs.sessions },
       } as const;

       export type LogType = keyof typeof LOG_TYPE_MAP;

       function getLogFilePath(type: LogType, date?: Date): string {
         const dateStr = (date ?? new Date()).toISOString().slice(0, 10); // YYYY-MM-DD per D-04
         return join(LOG_TYPE_MAP[type].dir, `${dateStr}.jsonl`);
       }

       export async function appendLogEntry(type: LogType, rawEntry: unknown): Promise<void> {
         await ensureInit();
         const schema = LOG_TYPE_MAP[type].schema;
         const validated = (schema as z.ZodType).parse(rawEntry);
         const scrubbed = scrubObject(validated); // D-01: scrub before write
         const line = JSON.stringify(scrubbed) + '\n';
         const filePath = getLogFilePath(type);
         await appendFile(filePath, line, 'utf-8'); // Native appendFile, NOT write-file-atomic (see anti-patterns)
       }
       ```

       Key design decisions:
       - Uses native `fs.appendFile` NOT `write-file-atomic` for JSONL appends (RESEARCH.md anti-pattern: write-file-atomic replaces entire file, appendFile is atomic for <4KB on POSIX)
       - Validates via Zod schema before scrubbing (catches bad data early)
       - Scrubs after validation (ensures all validated string fields are cleaned)
       - Daily rotation is automatic via date-based filename (D-04)
       - `ensureInit()` called at start to create directories (Pattern 1)
